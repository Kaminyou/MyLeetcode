### Combinations with module
```cpp
long long mod = 1e9 + 7;
long long inv(long long a) {
    if (a == 1) return 1;
    return (mod - mod / a) * inv(mod % a) % mod;
}
int comb(int n, int k) {
    long long res = 1L;
    for (int i = 0; i < k; ++i) {
        res = res * (n - i) % mod;
        res = res * inv(i + 1) % mod;
    } 
    return (int)(res % mod);
}
```
